function displayMessage() { if ('$mode' == 'newObject') { parent.setSaved('edit', '$object->getId()'); } top.displayMessage("Your object '$object->getMetaName()' is saved!"); top.reloadTreeFolder( $object->getMetaParentId() ); if ( parent.closeWindowOnSave ) { top.closeFrame(parent.window.name); } else { parent.setFormChanged(false); } }
